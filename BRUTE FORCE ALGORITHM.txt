1) Bubble Sort

#include<iostream>
using namespace std;
int main()
{
   int n,temp,flag;
   cout<<"Enter no of elements:\n";
   cin>>n;
   int a[n];
   cout<<"Enter elements:\n";
   for(int i=0;i<n;i++)
   {
      cin>>a[i];
   }
   for(int i=0;i<n-1;i++)
   {
      bool flag=false;
      for(int j=0;j<n-1-i;j++)
      {
         if(a[j]>a[j+1])
         {
            temp=a[j];
            a[j]=a[j+1];
            a[j+1]=temp;
            flag=true;
         }
      }
         if(!flag)
         {
            break;
         }
   }
   cout<<"Sorted array using bubble sorting:\n";
   for(int i=0;i<n;i++)
   {
      cout<<a[i]<<" ";
   }
   return 0;
}

2) Selection Sort :-

#include<iostream>
using namespace std;
int main()
{
   int n,temp,min;
   cout<<"Enter number of elements:";
   cin>>n;
   int a[n];
   cout<<"Enter elements:";
   for(int i=0;i<n;i++)
   {
      cin>>a[i];
   }
   for(int i=0;i<n-1;i++)
   {
      min=i;
      for(int j=i+1;j<n;j++)
      {
         if (a[j]<a[min])
         {
            min=j;
         }
      }
      temp=a[i];
      a[i]=a[min];
      a[min]=temp;
   }
   cout<<"Sorted array using selection sort\n";
   for(int i=0;i<n;i++)
      cout<<a[i]<<"\t";
   return 0;
}

3)STRING SEARCHING :-

#include <iostream>
#include <string>
using namespace std;
int main() {
    string text, patt;
    cout << "Enter text: ";
    getline(cin, text);
    cout << "Enter pattern: ";
    getline(cin, patt);
    int n = 0, m = 0;
    while (text[n] != '\0') n++;
    while (patt[m] != '\0') m++;
    bool found = false;
    for (int i = 0; i <= n - m; i++) {
        int j = 0;
        while (j < m) {
            if (text[i + j] != patt[j]) {
                break;
            }
            j++;
        }
        if (j == m) {
            cout << "Pattern found at position: " << i << endl;
            found = true;
        }
    }
    if (!found) {
        cout << "Pattern not found" << endl;
    }
    return 0;
}

4)Closest Pair :-

#include<iostream>
#include<cmath>
#include<cfloat>
using namespace std;
int main()
{
   int n;
   cout<<"Enter no of points:";
   cin>>n;
   int x[n],y[n];
   for(int i=1;i<=n;i++)
   {
      cout<<"X-coordinates of "<<i<<endl;
      cin>>x[i];
      cout<<"Y-coordinates of "<<i<<endl;
      cin>>y[i];
   }
   float d=FLT_MAX;
   int p1=0,p2=0;
   for(int i=1;i<=n-1;i++)
   {
      for(int j=i+1;j<=n;j++)
      {
         double dist=sqrt(pow((x[i]-x[j]),2)+pow((y[i]-y[j]),2));
         if(dist<d)
         {
            d=dist;
            p1=i;
            p2=j;
         }
      }
   }
   cout<<"Closest pair:("<<x[p1]<<","<<y[p1]<<") and ("<<x[p2]<<","<<y[p2]<<")\n";
   cout<<"Minimum distance="<<d<<endl;
   return 0;
}
